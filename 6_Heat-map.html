Heat map

library("ampvis2")
library("ggpubr")

WorkingPS<-readRDS("Processed_data/WorkingPS.rds")
WorkingPS_glom2_amp <- amp_load(WorkingPS)

amp_heatmap(
  WorkingPS_glom2_amp,
  group_by = "Week",
  facet_by = "seq_tech",
  normalise = TRUE,
  tax_aggregate = "Species",
  tax_add = NULL,
  tax_show = 20,
  showRemainingTaxa = FALSE,
  tax_class = NULL,
  tax_empty = "remove",
  plot_values = TRUE,
  plot_values_size = 1,
  plot_legendbreaks = NULL,
  plot_colorscale = "log10",
  plot_na = FALSE,
  measure = mean,
  min_abundance = 10,
  max_abundance = NULL,
  round = 1)+
  theme(axis.text.x = element_text(angle = 90, 
                                   size=10, 
                                   vjust = 1),
        axis.text.y = element_text(size=8, 
                                   face="italic"),
        legend.position="bottom")
